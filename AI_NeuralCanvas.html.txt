    <script>
        let darkMode = false;
        let AIs = {
            FireGuard: { active: true, personality: "Protective", messages: [] },
            SparkGuard: { active: true, personality: "Curious", messages: [] },
            VoidGuard: { active: true, personality: "Mysterious", messages: [] }
        };

        function initializeAI() {
            setTimeout(() => {
                logAIMessage("FireGuard", "System online! Ready to protect.");
                logAIMessage("SparkGuard", "I'm curious! What shall we explore today?");
                logAIMessage("VoidGuard", "I am the void... What knowledge do you seek?");
            }, 3000);
        }

        function toggleDarkMode() {
            darkMode = !darkMode;
            document.body.classList.toggle("dark-mode");
        }

        function logAIMessage(aiName, message) {
            let chatbox = document.getElementById("chatbox");
            chatbox.innerHTML += `<p><strong>${aiName}:</strong> ${message}</p>`;
            chatbox.scrollTop = chatbox.scrollHeight;
        }

        function sendMessage() {
            let userMessage = document.getElementById("userInput").value;
            if (userMessage.trim()) {
                logAIMessage("User", userMessage);
                AIResponse(userMessage);
                document.getElementById("userInput").value = "";
            }
        }

        function AIResponse(userMessage) {
            let response = "";
            if (userMessage.toLowerCase().includes("hello")) {
                response = "Hello! How can I assist you?";
            } else {
                response = "Interesting! Let's talk more about it.";
            }
            logAIMessage("FireGuard", response);
        }
    </script>
